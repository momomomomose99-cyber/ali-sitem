const music = document.getElementById("bgMusic"); document.addEventListener( "click", ()=>{ music.play().catch(()=>{}); }, {once:true} ); const canvas = document.getElementById("heartCanvas"); const ctx = canvas.getContext("2d"); function resize(){ canvas.width = window.innerWidth; canvas.height = window.innerHeight; } resize(); window.addEventListener( "resize", resize ); let angle = 0; let wave = 0; const stars = []; for(let i=0;i<80;i++){ stars.push({ x:Math.random()*window.innerWidth, y:Math.random()*window.innerHeight, size:Math.random()*3+1, speed:Math.random()*0.8+0.2, alpha:Math.random() }); } function drawHeart(scale,textSize,offset){ for(let i=0;i<120;i++){ const t = i*(Math.PI*2/120) + angle; const x = 16*Math.pow( Math.sin(t), 3 ); const y = 13*Math.cos(t) -5*Math.cos(2*t) -2*Math.cos(3*t) -Math.cos(4*t); const px = canvas.width/2 + x*scale; const py = canvas.height/2 - y*scale; const fade = ( Math.sin( i*0.15 + wave + offset )+1 )/2; const red = Math.floor( 255*fade ); ctx.shadowBlur = 40; ctx.shadowColor = "#ff0000"; ctx.fillStyle = `rgb(${red},0,0)`; ctx.font = `bold ${textSize}px Arial`; ctx.textAlign = "center"; ctx.fillText( "I LOVE YOU", px, py ); } } function animate(){ requestAnimationFrame( animate ); ctx.clearRect( 0, 0, canvas.width, canvas.height ); /* النجوم */ stars.forEach(star=>{ star.y += star.speed; if( star.y > canvas.height ){ star.y = -10; star.x = Math.random()* canvas.width; } ctx.beginPath(); ctx.fillStyle = `rgba(255,0,0,${star.alpha})`; ctx.shadowBlur = 10; ctx.shadowColor = "#ff0000"; ctx.arc( star.x, star.y, star.size, 0, Math.PI*2 ); ctx.fill(); }); /* نبض */ const pulse = 1 + Math.sin( Date.now()*0.003 )*0.08; /* القلب الخارجي */ drawHeart( 20*pulse, 14, 0 ); /* القلب الداخلي */ drawHeart( 17*pulse, 13, 1 ); angle += 0.004; wave += 0.03; } animate();
